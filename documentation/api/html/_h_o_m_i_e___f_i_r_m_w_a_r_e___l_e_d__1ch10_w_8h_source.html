41 bool dimmerHandler(const HomieRange& range, const String& value, MyHomieNode* homieNode, MyHomieProperty* homieProperty) {
42  bool done = false; // set to true if value was processed and no further tasks necessary
43  if (homieProperty->isTarget("switch")) {
44  homieNode->setOption("ch*",a_PWMdimmer::SCALE, (value.startsWith(F("true")) ? 1 : 0));
45  homieProperty->sendValue((value.startsWith(F("true")) ? true : false)); // acknowledge value as successfully processed
46  done = true;
47  }
48  if (homieProperty->isTarget("dimmer")) {
49  float scale = value.toFloat()/100.0f;
50  if (scale>1) scale = 1.0f;
51  if (scale<0) scale = 0.0f;
52  homieNode->setOption("ch*",a_PWMdimmer::SCALE, scale);
53  homieProperty->sendValue(scale*100);
54  done = true;
55  }
56  return done; // unprocessed properties will be handled by the default handler
57 }

87  ->addProperty({"frequency","Frequency","hz", DATATYPE_INTEGER, RETAINED, "100:40000",SETTABLE,1,0,0,0})
88  ->registerInputHandler(&settingsHandler);
89 
90  // set default options
91  myDevice.setOption("DIMMERS","ch*",a_PWMdimmer::PWM_FREQUENCY , 100); // 100hz PWM Frequency as PWM on a ESP8266 is software generated. 1Khz produces some random flicker (WiFi?)
92  myDevice.setOption("DIMMERS","ch*",a_PWMdimmer::PWM_FADE_DURATION_MS , 100); // 100ms transition time for soft start/stop
