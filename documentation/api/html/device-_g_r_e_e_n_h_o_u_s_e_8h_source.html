104 bool humidifierInputHandler(const HomieRange& range, const String& value, MyHomieNode* homieNode, MyHomieProperty* homieProperty) {
105  bool success = false;
106  int option = enumGetIndex(String(homieProperty->getDef().format),value);
107  myLog.printf(LOG_INFO,F(" humidifierInputHandler received %s/%s = %d"), value.c_str(), homieProperty->getId(), option);
108  bool switchLight = false;
109  if (strcmp(homieProperty-> getId(),"humiLight")==0) {
110  switchLight = true;
111  }
112  switch (option)
113  {
114  case 0: // off
115  homieNode->getPropertyPlugin("humiPower")->set(LOW);
116  success = true;
117  break;
118  case 1: // on
119  if (!sequencer.sequenceRunning()) {
120  if (!switchLight) {
121  myLog.print(LOG_INFO,F("Starting humidifier on sequence"));
122  sequencer.startSequence(humidifierOn,&humidifierCallback,&doneCallback);
123  } else {
124  if (lightState==0) {
125  sequencer.startSequence(singleLongPulse,&humidifierCallback);
126  } if (lightState==2) {
127  sequencer.startSequence(doubleLongPulse,&humidifierCallback);
128  }
129  lightState = 1;
130  }
131  success = true;
132  }
133  break;
134  case 2: // pulse
135  if (!sequencer.sequenceRunning()) {
136  if (!switchLight) {
137  sequencer.startSequence(humidifierPulse,&humidifierCallback, &doneCallback);
138  } else {
139  if (lightState==0) {
140  sequencer.startSequence(doubleLongPulse,&humidifierCallback);
141  } if (lightState==1) {
142  sequencer.startSequence(singleLongPulse,&humidifierCallback);
143  }
144  lightState = 2;
145  }
146  success = true;
147  }
148  break;
149  default:
150  break;
151  }
152  if (success) {
153  homieProperty->sendValue((float) option);
154  }
155  return true;
156 };
