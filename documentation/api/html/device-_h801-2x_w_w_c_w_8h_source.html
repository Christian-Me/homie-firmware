42 bool pwmSettingsHandler(const HomieRange& range, const String& value, MyHomieNode* homieNode, MyHomieProperty* homieProperty) {
43  MyHomieNode *nodeH801 = myDevice.getNode("H801");
44  myLog.setAppName("H801");
45  if (nodeH801==NULL) return false;
46  myLog.printf(LOG_INFO,F(" H801 pwmSettings %s/%s = '%s'"), homieNode->getId(), homieProperty->getId(), value.c_str());
47  if (homieProperty->isTarget("frequency")) {
48  for (uint8_t i=0; i < nodeH801->length(); i++) { // use the same settings for all channels
49  nodeH801->getProperty(i)->getPlugin()->setOption(a_PWMdimmer::PWM_FREQUENCY, (int) value.toInt());
50  }
51  homieProperty->sendValue(value.toFloat()); // confirm directly as sample rate = 0;
52  } else
53  if (homieProperty->isTarget("duration")) {
54  for (uint8_t i=0; i < nodeH801->length(); i++) {
55  nodeH801->getProperty(i)->getPlugin()->setOption(a_PWMdimmer::PWM_FADE_DURATION_S , (int) value.toInt());
56  }
57  homieProperty->sendValue(value.toFloat());
58  } else
59  if (homieProperty->isTarget("shortestDelay")) {
60  for (uint8_t i=0; i < nodeH801->length(); i++) {
61  nodeH801->getProperty(i)->getPlugin()->setOption(a_PWMdimmer::PWM_FADE_DELAY , (int) value.toInt());
62  }
63  homieProperty->sendValue(value.toFloat());
64  } else
65  {
66  return false;
67  }
68  return true;

77 bool tuneableHandler(const HomieRange& range, const String& value, MyHomieNode* homieNode, MyHomieProperty* homieProperty) {
78  MyHomieNode *dimmerNode = myDevice.getNode("H801");
79  Plugin *plugin = dimmerNode->plugin;
80  myLog.printf(LOG_INFO,F(" tuneableHandler %s/%s Plugin:%s '%s'"), homieNode->getId(), homieProperty->getId(), plugin->id(), value.c_str());
81  float ww = 0;
82  float cw = 0;
83  int channel = atoi(homieProperty->getId())-1;
84  if (homieProperty->isTarget("temp",2)) {
85  temperature[channel] = (value.toFloat()-3000) * 0.0666; //scale 3000-6000 to 0-200
86  } else
87  if (homieProperty->isTarget("dimmer",2)) {
88  dimmer[channel] = value.toFloat()/100;
89  } else
90  if (homieProperty->isTarget("switch",2)) {
91  factor[channel] = (value.equalsIgnoreCase("true")) ? 1 : 0;
92  } else
93  { // no property handled here
94  return false;
95  }
96 
97  if (temperature[channel]<100) {
98  ww=100;
99  cw=temperature[channel];
100  } else {
101  ww=200-temperature[channel];
102  cw=100;
103  }
104  dimmerNode->setValue(channelsWW[channel], ww*dimmer[channel]*factor[channel]);
105  dimmerNode->setValue(channelsCW[channel], cw*dimmer[channel]*factor[channel]);
106  homieProperty->sendValue(value.c_str());
107  return true;
108 };
