114 bool ADS1115readHandler(uint8_t task, MyHomieNode* homieNode, MyHomieProperty* homieProperty) {
115  bool result = true;
116  switch (task) {
117  case TASK_BEFORE_SAMPLE: {
118  // switch on resistive soil sensor for measurement
119  timer = millis();
120  myLog.printf(LOG_TRACE,F(" before sample %s/%s (%d)"),homieNode->getId(),homieProperty->getId(),timer);
121  digitalWrite(D8,HIGH);
122  delay(500); // not nice! Will find a solution later (extra timer?)
123  break;
124  };
125  case TASK_AFTER_READ: {
126  // switch off resistive soil sensor to avoid corrosion
127  myLog.printf(LOG_TRACE,F(" after read %s/%s on for %dms (%d)"),homieNode->getId(),homieProperty->getId(),millis()-timer,timer);
128  //drop first reading
129  if (firstReading) {
130  firstReading = false;
131  result = false;
132  }
133  // check for spikes > 500mV but not on boot where value==0;
134  if (homieProperty->getValue()!=0 && fabs(homieProperty->getReadValue()-homieProperty->getValue()) > 500) {
135  myLog.printf(LOG_INFO,F(" %s/%s spike detected %.2fmv"),homieNode->getId(),homieProperty->getId(),homieProperty->getReadValue()-homieProperty->getValue());
136  result = false;
137  spikeCounter++;
138  if (spikeCounter>2) {
139  result=true;
140  spikeCounter=0;
141  }
142  }
143  digitalWrite(D8,LOW);
144  break;
145  };
146  };
147 
148  return result;
149 };
