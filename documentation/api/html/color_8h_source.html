50 uint32_t colorHSV2RGB(float hue, uint8_t sat, uint8_t val) {
51  int i;
52  float f,p,q,t;
53  float h, s, v;
54  packedColor color;
55 
56  //expand the u8 hue in range 0->255 to 0->359* (there are problems at exactly 360)
57  // h = 359.0 * ((float)hsv->h / 255.0);
58  h = hue;
59  if (h>=360) h=359.9;
60 
61  h = MAX(0.0, MIN(360.0, h));
62  s = MAX(0.0, MIN(100.0, sat));
63  v = MAX(0.0, MIN(100.0, val));
64 
65  s /= 100;
66  v /= 100;
67  if(s == 0) {
68  // Achromatic (grey)
69  color.value.r = color.value.g = color.value.b = round(v*255);
70  return color.color;
71  }
72 
73  h /= 60; // sector 0 to 5
74  i = floor(h);
75  f = h - i; // factorial part of h
76  p = v * (1 - s);
77  q = v * (1 - s * f);
78  t = v * (1 - s * (1 - f));
79  switch(i) {
80  case 0:
81  color.value.r = round(255*v);
82  color.value.g = round(255*t);
83  color.value.b = round(255*p);
84  break;
85  case 1:
86  color.value.r = round(255*q);
87  color.value.g = round(255*v);
88  color.value.b = round(255*p);
89  break;
90  case 2:
91  color.value.r = round(255*p);
92  color.value.g = round(255*v);
93  color.value.b = round(255*t);
94  break;
95  case 3:
96  color.value.r = round(255*p);
97  color.value.g = round(255*q);
98  color.value.b = round(255*v);
99  break;
100  case 4:
101  color.value.r = round(255*t);
102  color.value.g = round(255*p);
103  color.value.b = round(255*v);
104  break;
105  default: // case 5:
106  color.value.r = round(255*v);
107  color.value.g = round(255*p);
108  color.value.b = round(255*q);
109  }

128 uint32_t colorHSV(uint16_t hue, uint8_t sat, uint8_t val) {
129 
130  uint8_t r, g, b;
131 
132  // Remap 0-65535 to 0-1529. Pure red is CENTERED on the 64K rollover;
133  // 0 is not the start of pure red, but the midpoint...a few values above
134  // zero and a few below 65536 all yield pure red (similarly, 32768 is the
135  // midpoint, not start, of pure cyan). The 8-bit RGB hexcone (256 values
136  // each for red, green, blue) really only allows for 1530 distinct hues
137  // (not 1536, more on that below), but the full unsigned 16-bit type was
138  // chosen for hue so that one's code can easily handle a contiguous color
139  // wheel by allowing hue to roll over in either direction.
140  hue = (hue * 1530L + 32768) / 65536;
141  // Because red is centered on the rollover point (the +32768 above,
142  // essentially a fixed-point +0.5), the above actually yields 0 to 1530,
143  // where 0 and 1530 would yield the same thing. Rather than apply a
144  // costly modulo operator, 1530 is handled as a special case below.
145 
146  // So you'd think that the color "hexcone" (the thing that ramps from
147  // pure red, to pure yellow, to pure green and so forth back to red,
148  // yielding six slices), and with each color component having 256
149  // possible values (0-255), might have 1536 possible items (6*256),
150  // but in reality there's 1530. This is because the last element in
151  // each 256-element slice is equal to the first element of the next
152  // slice, and keeping those in there this would create small
153  // discontinuities in the color wheel. So the last element of each
154  // slice is dropped...we regard only elements 0-254, with item 255
155  // being picked up as element 0 of the next slice. Like this:
156  // Red to not-quite-pure-yellow is: 255, 0, 0 to 255, 254, 0
157  // Pure yellow to not-quite-pure-green is: 255, 255, 0 to 1, 255, 0
158  // Pure green to not-quite-pure-cyan is: 0, 255, 0 to 0, 255, 254
159  // and so forth. Hence, 1530 distinct hues (0 to 1529), and hence why
160  // the constants below are not the multiples of 256 you might expect.
161 
162  // Convert hue to R,G,B (nested ifs faster than divide+mod+switch):
163  if(hue < 510) { // Red to Green-1
164  b = 0;
165  if(hue < 255) { // Red to Yellow-1
166  r = 255;
167  g = hue; // g = 0 to 254
168  } else { // Yellow to Green-1
169  r = 510 - hue; // r = 255 to 1
170  g = 255;
171  }
172  } else if(hue < 1020) { // Green to Blue-1
173  r = 0;
174  if(hue < 765) { // Green to Cyan-1
175  g = 255;
176  b = hue - 510; // b = 0 to 254
177  } else { // Cyan to Blue-1
178  g = 1020 - hue; // g = 255 to 1
179  b = 255;
180  }
181  } else if(hue < 1530) { // Blue to Red-1
182  g = 0;
183  if(hue < 1275) { // Blue to Magenta-1
184  r = hue - 1020; // r = 0 to 254
185  b = 255;
186  } else { // Magenta to Red-1
187  r = 255;
188  b = 1530 - hue; // b = 255 to 1
189  }
190  } else { // Last 0.5 Red (quicker than % operator)
191  r = 255;
192  g = b = 0;
193  }
194 
195  // Apply saturation and value to R,G,B, pack into 32-bit result:
196  uint32_t v1 = 1 + val; // 1 to 256; allows >>8 instead of /255
197  uint16_t s1 = 1 + sat; // 1 to 256; same reason
198  uint8_t s2 = 255 - sat; // 255 to 0
199  return ((((((r * s1) >> 8) + s2) * v1) & 0xff00) << 8) |
200  (((((g * s1) >> 8) + s2) * v1) & 0xff00) |
201  ( ((((b * s1) >> 8) + s2) * v1) >> 8);
202 }
